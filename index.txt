        function exportData() {
            try {
                const dataStr = JSON.stringify(systemData, null, 2);
                const blob = new Blob([dataStr], { type: 'application/json' });
                const url = window.URL.createObjectURL(blob);
                const link = document.createElement('a');
                link.href = url;
                
                const now = new Date();
                const year = now.getFullYear();
                const month = String(now.getMonth() + 1).padStart(2, '0');
                const day = String(now.getDate()).padStart(2, '0');
                const hour = String(now.getHours()).padStart(2, '0');
                const minute = String(now.getMinutes()).padStart(2, '0');
                
                link.download = `MCOT_Data_${year}${month}${day}_${hour}${minute}.json`;
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                window.URL.revokeObjectURL(url);
                
                showNotification('ส่งออกข้อมูลสำเร็จ', 'success');
            } catch (error) {
                console.error('Export error:', error);
                showNotification('เกิดข้อผิดพลาดในการส่งออกข้อมูล', 'error');
            }
        }
        
        function importData(event) {
            const file = event.target.files[0];
            if (!file) return;
            
            if (!file.name.toLowerCase().endsWith('.json')) {
                showNotification('กรุณาเลือกไฟล์ .json เท่านั้น', 'error');
                event.target.value = '';
                return;
            }
            
            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const importedData = JSON.parse(e.target.result);
                    
                    if (!importedData || typeof importedData !== 'object') {
                        showNotification('ไฟล์ข้อมูลไม่ถูกต้อง', 'error');
                        event.target.value = '';
                        return;
                    }
                    
                    if (confirm("คุณแน่ใจหรือไม่ว่าต้องการนำเข้าข้อมูล?\n\nข้อมูลปัจจุบันจะถูกเขียนทับทั้งหมด")) {
                        systemData = importedData;
                        
                        if (!systemData.customers) systemData.customers = [];
                        if (!systemData.products) systemData.products = [];
                        if (!systemData.packages) systemData.packages = [];
                        if (!systemData.orders) systemData.orders = [];
                        if (!systemData.settings) {
                            systemData.settings = { 
                                lastSave: new Date().toISOString(), 
                                version: "4.0", 
                                sidebarExpanded: false,
                                lastTab: 'customers'
                            };
                        }
                        
                        updateProductCodeNumbers();
                        updatePackageCounter();
                        
                        if (systemData.settings.sidebarExpanded) {
                            const sidebar = document.getElementById('sidebar');
                            const mainContent = document.getElementById('main-content');
                            if (sidebar && mainContent) {
                                sidebar.classList.add('expanded');
                                mainContent.classList.add('expanded');
                                isSidebarExpanded = true;
                            }
                        }
                        
                        saveData();
                        
                        const activeTab = systemData.settings.lastTab || 'customers';
                        switchTab(activeTab);
                        
                        updateBadges();
                        showNotification('นำเข้าข้อมูลสำเร็จ', 'success');
                    }
                } catch (parseError) {
                    console.error('JSON parse error:', parseError);
                    showNotification('ไม่สามารถอ่านไฟล์ JSON ได้', 'error');
                }
                event.target.value = '';
            };
            reader.readAsText(file);
        }

        // ========== PRODUCT CODE ==========
        function updateProductCodeNumbers() {
            productTypeCodes.onAir.nextNumber = 1;
            productTypeCodes.online.nextNumber = 1;
            productTypeCodes.production.nextNumber = 1;
            systemData.products.forEach(product => {
                if (product.code) {
                    const prefix = product.code.split('-')[0];
                    const number = parseInt(product.code.split('-')[1]);
                    if (prefix === 'OA' && number >= productTypeCodes.onAir.nextNumber) productTypeCodes.onAir.nextNumber = number + 1;
                    else if (prefix === 'OL' && number >= productTypeCodes.online.nextNumber) productTypeCodes.online.nextNumber = number + 1;
                    else if (prefix === 'Pr' && number >= productTypeCodes.production.nextNumber) productTypeCodes.production.nextNumber = number + 1;
                }
            });
        }
        
        function updatePackageCounter() {
            if (systemData.packages && systemData.packages.length > 0) {
                let maxNumber = 0;
                systemData.packages.forEach(pkg => {
                    if (pkg.code) {
                        const match = pkg.code.match(/^PKG-(\d+)$/);
                        if (match) {
                            const number = parseInt(match[1]);
                            if (number > maxNumber) maxNumber = number;
                        }
                    }
                });
                packageCounter = maxNumber + 1;
            } else {
                packageCounter = 1;
            }
        }
        
        function generateProductCode(type) {
            if (!type || !productTypeCodes[type]) return '';
            const codeInfo = productTypeCodes[type];
            const codeNumber = codeInfo.nextNumber.toString().padStart(3, '0');
            const code = `${codeInfo.prefix}-${codeNumber}`;
            productTypeCodes[type].nextNumber++;
            return code;
        }
        
        function generatePackageCode() {
            const codeNumber = packageCounter.toString().padStart(3, '0');
            const code = `PKG-${codeNumber}`;
            packageCounter++;
            return code;
        }

        function addProduct(event) {
            if (event) event.preventDefault();
            const name = document.getElementById('product-name');
            const price = document.getElementById('product-price');
            const type = document.getElementById('product-type');
            if (!name || !price || !type) return;
            const nameVal = name.value;
            const priceVal = price.value;
            const typeVal = type.value;
            const productId = document.getElementById('product-id').value;
            if (!nameVal || !priceVal || !typeVal) {
                showNotification('กรุณากรอกข้อมูลที่จำเป็นให้ครบถ้วน', 'error');
                return;
            }
            const productData = {
                id: productId || `PROD-${Date.now()}`,
                code: document.getElementById('product-code').value || generateProductCode(typeVal),
                name: nameVal,
                type: typeVal,
                price: parseFloat(priceVal),
                promotion: document.getElementById('product-promotion').value,
                promotionDetail: document.getElementById('product-promotion').value === 'มี' ? document.getElementById('product-promotion-detail').value : undefined,
                sampleLink: document.getElementById('product-sample-link').value || '',
                note: document.getElementById('product-note').value,
                created: productId ? systemData.products.find(p => p.id === productId)?.created : new Date().toISOString(),
                updated: new Date().toISOString()
            };
            if (productId) {
                const index = systemData.products.findIndex(p => p.id === productId);
                if (index !== -1) { systemData.products[index] = productData; showNotification('แก้ไขสินค้าเรียบร้อยแล้ว', 'success'); }
            } else {
                systemData.products.push(productData);
                showNotification('เพิ่มสินค้าเรียบร้อยแล้ว', 'success');
            }
            saveData();
            renderProducts();
            resetProductForm();
            hideProductForm();
        }

        function calculateOrderTotal() {
            // ดึงข้อมูลจากฟอร์ม
            const productSelect = document.getElementById('order-product');
            const selectedOption = productSelect?.selectedOptions[0];
            const unitPrice = parseFloat(selectedOption?.dataset.price) || 0;
            const timesPerDay = currentOrder.selectedSlots.length || 1;
            const startDate = new Date(currentOrder.startDate);
            const endDate = new Date(currentOrder.endDate);
            
            // คำนวณจำนวนวัน
            let totalDays = 0;
            if (startDate && endDate && endDate >= startDate) {
                const diffTime = Math.abs(endDate - startDate);
                const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)) + 1;
                totalDays = diffDays;
            }
            
            const basePrice = unitPrice * timesPerDay * totalDays;
            let currentPrice = basePrice;
            
            // ส่วนลดพิเศษ
            let specialPercent = 0;
            let specialAmount = 0;
            if (document.getElementById('specialDiscountToggle')?.checked) {
                specialPercent = parseFloat(document.getElementById('specialDiscountPercent')?.value) || 0;
                specialAmount = currentPrice * (specialPercent / 100);
                currentPrice -= specialAmount;
            }
            
            // ส่วนลดหน่วยงานราชการ (40% อัตโนมัติถ้าลูกค้าเป็นราชการ)
            let govPercent = 0;
            let govAmount = 0;
            const selectedCustomer = systemData.customers.find(c => c.id === currentOrder.selectedCustomerId);
            if (selectedCustomer && selectedCustomer.type === 'ราชการ') {
                govPercent = 40;
                govAmount = currentPrice * (govPercent / 100);
                currentPrice -= govAmount;
            }
            
            // ส่วนลดเงื่อนไข
            let conditionPercent = 0;
            let conditionAmount = 0;
            const conditionSelect = document.getElementById('conditionDiscount');
            if (conditionSelect) {
                conditionPercent = parseFloat(conditionSelect.value) || 0;
                conditionAmount = currentPrice * (conditionPercent / 100);
                currentPrice -= conditionAmount;
            }
            
            // ส่วนลดจ่ายล่วงหน้า
            let prepaymentAmount = 0;
            if (document.getElementById('prepaymentDiscount')?.checked) {
                prepaymentAmount = currentPrice * 0.05;
                currentPrice -= prepaymentAmount;
            }
            
            const afterDiscount = currentPrice;
            const vat = afterDiscount * 0.07;
            const afterVat = afterDiscount + vat;
            
            // ภาษีหัก ณ ที่จ่าย
            let whtPercent = 0;
            let whtAmount = 0;
            if (document.getElementById('whtToggle')?.checked) {
                whtPercent = parseFloat(document.getElementById('whtPercent')?.value) || 0;
                whtAmount = afterDiscount * (whtPercent / 100);
            }
            
            const netTotal = afterVat - whtAmount;
            const totalDiscount = specialAmount + govAmount + conditionAmount + prepaymentAmount;
            
            // อัปเดต UI (ใช้ formatNumber)
            document.getElementById('displayBasePrice').textContent = formatNumber(basePrice) + ' ฿';
            document.getElementById('displaySpecialPercent').textContent = specialPercent;
            document.getElementById('displaySpecialAmount').textContent = formatNumber(specialAmount) + ' ฿';
            document.getElementById('displayGovAmount').textContent = formatNumber(govAmount) + ' ฿';
            document.getElementById('displayConditionPercent').textContent = conditionPercent;
            document.getElementById('displayConditionAmount').textContent = formatNumber(conditionAmount) + ' ฿';
            document.getElementById('displayPrepaymentAmount').textContent = formatNumber(prepaymentAmount) + ' ฿';
            document.getElementById('displayTotalDiscount').textContent = formatNumber(totalDiscount) + ' ฿';
            document.getElementById('displayAfterDiscount').textContent = formatNumber(afterDiscount) + ' ฿';
            document.getElementById('displayVat').textContent = formatNumber(vat) + ' ฿';
            document.getElementById('displayTotalWithVat').textContent = formatNumber(afterVat) + ' ฿';
            document.getElementById('displayWhtPercent').textContent = whtPercent;
            document.getElementById('displayWhtAmount').textContent = formatNumber(whtAmount) + ' ฿';
            document.getElementById('displayNetTotal').textContent = formatNumber(netTotal) + ' ฿';
            
            // แสดง/ซ่อน summary
            document.getElementById('summarySpecialDiscount').classList.toggle('hidden', !document.getElementById('specialDiscountToggle')?.checked || specialAmount === 0);
            document.getElementById('summaryGovDiscount').classList.toggle('hidden', govAmount === 0);
            document.getElementById('summaryWHT').classList.toggle('hidden', !document.getElementById('whtToggle')?.checked || whtAmount === 0);
            
            // แสดงผู้อนุมัติ
            const approverText = getApproverText();
            document.getElementById('displayApprover').textContent = approverText ? 'อนุมัติโดย: ' + approverText : '';
            
            return {
                basePrice,
                specialPercent,
                specialAmount,
                govPercent,
                govAmount,
                conditionPercent,
                conditionAmount,
                prepaymentAmount,
                totalDiscount,
                afterDiscount,
                vat,
                afterVat,
                whtPercent,
                whtAmount,
                netTotal,
                totalDays
            };
        }
        
        function addOrderToCampaign() {
            const customerId = document.getElementById('order-customer').value;
            const customer = systemData.customers.find(c => c.id === customerId);
            
            if (!customer) {
                showNotification('กรุณาเลือกลูกค้า', 'error');
                return;
            }
            
            const productSelect = document.getElementById('order-product');
            const productId = productSelect?.value;
            let product = null;
            
            if (currentOrder.orderType === 'Package') {
                product = systemData.packages.find(p => p.id === productId);
            } else {
                product = systemData.products.find(p => p.id === productId);
            }
            
            if (!product) {
                showNotification('กรุณาเลือกสินค้า', 'error');
                return;
            }
            
            const calc = calculateOrderTotal();
            
            const newOrder = {
                id: Date.now(),
                date: new Date().toISOString(),
                customerId: customer.id,
                customerName: customer.company,
                productName: product.name,
                productCode: product.code || '',
                unitPrice: product.price || 0,
                timesPerDay: currentOrder.selectedSlots.length || 1,
                totalDays: calc.totalDays || 0,
                basePrice: calc.basePrice,
                specialDiscount: {
                    enabled: document.getElementById('specialDiscountToggle')?.checked || false,
                    percent: calc.specialPercent,
                    amount: calc.specialAmount,
                    approver: getApproverText()
                },
                govDiscount: {
                    enabled: customer.type === 'ราชการ',
                    percent: calc.govPercent,
                    amount: calc.govAmount
                },
                conditionDiscount: {
                    percent: calc.conditionPercent,
                    amount: calc.conditionAmount
                },
                prepaymentDiscount: document.getElementById('prepaymentDiscount')?.checked || false,
                totalDiscount: calc.totalDiscount,
                afterDiscount: calc.afterDiscount,
                vat: calc.vat,

        function updateCampaignTotal() {

        // ========== QUOTATION ==========
        function populateQuotationCustomerSelect() {
            const select = document.getElementById('quotation-customer');
            if (!select) return;
            select.innerHTML = '<option value="">-- เลือกลูกค้า --</option>';
            systemData.customers.forEach(c => {
                const opt = document.createElement('option');
                opt.value = c.id;
                opt.textContent = c.company;
                select.appendChild(opt);
            });
        }
        
        function loadCustomerQuotation() {
            const customerId = document.getElementById('quotation-customer').value;
            if (!customerId) {
                document.getElementById('quotation-preview').classList.add('hidden');
                document.getElementById('quotation-empty').classList.remove('hidden');
                return;
            }
            
            const customer = systemData.customers.find(c => c.id === customerId);
            const customerOrders = systemData.orders.filter(o => o.customerId === customerId);
            
            if (!customer || customerOrders.length === 0) {
                document.getElementById('quotation-preview').classList.add('hidden');
                document.getElementById('quotation-empty').classList.remove('hidden');
                return;
            }
            
            let beforeVat = 0, vat = 0, wht = 0, total = 0;
            customerOrders.forEach(o => {
                beforeVat += o.afterDiscount || 0;
                vat += o.vat || 0;
                wht += o.wht?.amount || 0;
                total += o.netTotal || 0;
            });
            
            document.getElementById('quotation-company').textContent = customer.company;
            document.getElementById('quotation-address').textContent = customer.address || '-';
            document.getElementById('quotation-phone').textContent = customer.phone || '-';
            document.getElementById('quotation-date').textContent = new Date().toLocaleDateString('th-TH', { day: '2-digit', month: 'long', year: 'numeric' });
            document.getElementById('quotation-staff-name').textContent = document.getElementById('staff-name').value || 'ฝ่ายขาย';
            
            const itemsContainer = document.getElementById('quotation-items');
            itemsContainer.innerHTML = '';
            
            customerOrders.forEach((o, i) => {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td class="p-2">${i+1}. ${escapeHtml(o.productName || 'รายการ')}<br><small class="text-slate-400">${new Date(o.date).toLocaleDateString('th-TH')}</small></td>
                    <td class="p-2 text-center">${o.totalDays || 0}</td>
                    <td class="p-2 text-center">${o.timesPerDay || 1}</td>
                    <td class="p-2 text-right">${(o.totalDays * o.timesPerDay) || 0}</td>
                    <td class="p-2 text-right">${formatNumber(o.afterDiscount || 0)} ฿</td>
                    <td class="p-2 text-right">${formatNumber(o.wht?.amount || 0)} ฿</td>
                    <td class="p-2 text-right font-bold">${formatNumber(o.netTotal || 0)} ฿</td>
                `;
                itemsContainer.appendChild(row);
            });
            
            document.getElementById('quotation-before-vat').textContent = formatNumber(beforeVat) + ' บาท';
            document.getElementById('quotation-vat').textContent = formatNumber(vat) + ' บาท';
            document.getElementById('quotation-wht').textContent = formatNumber(wht) + ' บาท';
            document.getElementById('quotation-total').textContent = formatNumber(total) + ' บาท';
            
            document.getElementById('quotation-preview').classList.remove('hidden');
            document.getElementById('quotation-empty').classList.add('hidden');
        }

        function updateLastSaved() {
            const el = document.getElementById('last-saved');
            if (el) {
                if (systemData.settings.lastSave) {
                    const d = new Date(systemData.settings.lastSave);
                    el.textContent = `${d.toLocaleDateString('th-TH')} ${d.toLocaleTimeString('th-TH')}`;
                } else el.textContent = 'ยังไม่เคยบันทึก';
            }
        }
        
        function formatDate(d) { return new Date(d).toLocaleDateString('th-TH'); }
        
        function escapeHtml(unsafe) {
            if (!unsafe) return '';
            return String(unsafe)
                .replace(/&/g, "&amp;")
                .replace(/</g, "&lt;")
                .replace(/>/g, "&gt;")
                .replace(/"/g, "&quot;")
                .replace(/'/g, "&#039;");
        }
        
        function showNotification(message, type) {
            const old = document.querySelector('.notification');
            if (old) old.remove();
            const n = document.createElement('div');
            n.className = `notification fixed top-4 right-4 px-4 lg:px-6 py-2 lg:py-3 rounded-lg shadow-lg text-white font-medium z-50 animate-fadeIn ${type === 'success' ? 'bg-emerald-500' : type === 'error' ? 'bg-red-500' : 'bg-blue-500'}`;
            n.textContent = message;
            document.body.appendChild(n);
            setTimeout(() => {
                n.style.opacity = '0';
                n.style.transition = 'opacity 0.3s';
                setTimeout(() => n.remove(), 300);
            }, 3000);
        }
